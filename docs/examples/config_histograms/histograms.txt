# Example histogram configuration file
# This file demonstrates the various options for defining histograms from config

# Simple 1D histogram - only required fields
name=example_hist1 variable=example_var weight=example_weight bins=50 lowerBound=0.0 upperBound=10.0 suffix=simple

# Histogram with label
name=example_hist2 variable=example_var weight=example_weight bins=50 lowerBound=0.0 upperBound=10.0 label=Example Variable [units] suffix=labeled

# Histogram with channel dimension
name=example_hist3 variable=example_var weight=example_weight bins=50 lowerBound=0.0 upperBound=10.0 channelVariable=channel_var channelBins=2 channelLowerBound=0.0 channelUpperBound=2.0 channelRegions=channel1,channel2 suffix=with_channel
